# 第11章 图的遍历

## 11.1 深度优先搜索

### 深度优先搜索（DFS）基础概念

深度优先搜索（Depth-First Search，简称DFS）是一种用于遍历或搜索图或树的算法。它的基本思想是从起始顶点开始，沿着一条路径尽可能深入地访问顶点，直到无法继续为止，然后回溯到上一个有未访问邻接点的顶点，继续探索。

在图论中，DFS常用于：

- 检测图中的环
- 寻找连通分量
- 拓扑排序
- 解决迷宫问题
- 判断图是否为二分图

DFS可以使用递归或栈来实现。递归实现更加简洁直观，而栈实现则避免了递归可能导致的栈溢出问题。

### DFS实现方法表格

| 功能名称   | 实现方法                             | 具体说明                               |
| ---------- | ------------------------------------ | -------------------------------------- |
| 递归DFS    | dfs_recursive(graph, start, visited) | 使用递归方式实现，代码简洁但可能栈溢出 |
| 栈DFS      | dfs_iterative(graph, start)          | 使用显式栈实现，避免递归深度限制       |
| 完整图遍历 | dfs_complete(graph)                  | 遍历图中所有连通分量                   |

### 递归实现深度优先搜索

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

// 定义图的最大顶点数
#define MAX_VERTICES 100

// 图的邻接矩阵表示
int graph[MAX_VERTICES][MAX_VERTICES];
int numVertices;

// 访问标记数组
bool visited[MAX_VERTICES];

// 初始化图
void initGraph(int vertices) {
    // 设置顶点数量
    numVertices = vertices;
    // 初始化邻接矩阵为0（无边）
    for (int i = 0; i < numVertices; i++) {
        for (int j = 0; j < numVertices; j++) {
            graph[i][j] = 0;
        }
    }
    // 初始化访问标记为false
    for (int i = 0; i < numVertices; i++) {
        visited[i] = false;
    }
}

// 添加边（无向图）
void addEdge(int u, int v) {
    // 在邻接矩阵中设置边（无向图需要设置两个方向）
    graph[u][v] = 1;
    graph[v][u] = 1;
}

// 递归深度优先搜索
void dfsRecursive(int vertex) {
    // 标记当前顶点为已访问
    visited[vertex] = true;
    // 打印当前顶点
    printf("%d ", vertex);
    
    // 遍历所有邻接顶点
    for (int i = 0; i < numVertices; i++) {
        // 如果存在边且邻接顶点未被访问
        if (graph[vertex][i] == 1 && !visited[i]) {
            // 递归访问邻接顶点
            dfsRecursive(i);
        }
    }
}

// 对整个图进行DFS遍历（处理不连通图）
void dfsComplete() {
    // 重置访问标记
    for (int i = 0; i < numVertices; i++) {
        visited[i] = false;
    }
    
    // 遍历所有顶点
    for (int i = 0; i < numVertices; i++) {
        // 如果顶点未被访问，则从该顶点开始DFS
        if (!visited[i]) {
            printf("从顶点 %d 开始的连通分量: ", i);
            dfsRecursive(i);
            printf("\n");
        }
    }
}

// 主函数 - 演示DFS
int main() {
    // 初始化一个包含6个顶点的图
    initGraph(6);
    
    // 添加边构建图
    addEdge(0, 1);
    addEdge(0, 2);
    addEdge(1, 3);
    addEdge(2, 4);
    addEdge(3, 5);
    // 注意：这个图有两个连通分量：{0,1,2,3,4,5}（实际上是一个连通分量）
    // 让我们添加一个孤立顶点来演示多个连通分量
    // 顶点5已经连接，所以所有顶点都是连通的
    
    printf("深度优先搜索遍历结果:\n");
    dfsComplete();
    
    return 0;
}
```

### 栈实现深度优先搜索

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

// 定义栈结构
typedef struct {
    int items[MAX_VERTICES];
    int top;
} Stack;

// 初始化栈
void initStack(Stack* stack) {
    // 设置栈顶指针为-1（空栈）
    stack->top = -1;
}

// 判断栈是否为空
bool isEmpty(Stack* stack) {
    // 栈顶为-1表示空栈
    return stack->top == -1;
}

// 入栈操作
void push(Stack* stack, int value) {
    // 检查栈是否已满
    if (stack->top >= MAX_VERTICES - 1) {
        printf("栈溢出!\n");
        return;
    }
    // 增加栈顶指针并存储值
    stack->items[++(stack->top)] = value;
}

// 出栈操作
int pop(Stack* stack) {
    // 检查栈是否为空
    if (isEmpty(stack)) {
        printf("栈为空!\n");
        return -1;
    }
    // 返回栈顶元素并减少栈顶指针
    return stack->items[(stack->top)--];
}

// 使用栈实现的迭代DFS
void dfsIterative(int start, int graph[][MAX_VERTICES], int numVertices) {
    // 创建访问标记数组
    bool visited[MAX_VERTICES] = {false};
    // 创建栈
    Stack stack;
    initStack(&stack);
    
    // 将起始顶点入栈
    push(&stack, start);
    
    // 当栈不为空时继续
    while (!isEmpty(&stack)) {
        // 弹出栈顶顶点
        int current = pop(&stack);
        
        // 如果当前顶点未被访问
        if (!visited[current]) {
            // 标记为已访问
            visited[current] = true;
            // 打印顶点
            printf("%d ", current);
            
            // 将所有未访问的邻接顶点入栈
            // 注意：这里从后往前遍历是为了保持与递归版本相同的访问顺序
            for (int i = numVertices - 1; i >= 0; i--) {
                if (graph[current][i] == 1 && !visited[i]) {
                    push(&stack, i);
                }
            }
        }
    }
}

// 主函数 - 演示迭代DFS
int main() {
    // 创建图
    int graph[MAX_VERTICES][MAX_VERTICES] = {0};
    int numVertices = 6;
    
    // 添加边
    graph[0][1] = graph[1][0] = 1;
    graph[0][2] = graph[2][0] = 1;
    graph[1][3] = graph[3][1] = 1;
    graph[2][4] = graph[4][2] = 1;
    graph[3][5] = graph[5][3] = 1;
    
    printf("迭代深度优先搜索遍历结果 (从顶点0开始): ");
    dfsIterative(0, graph, numVertices);
    printf("\n");
    
    return 0;
}
```

深度优先搜索通过递归或栈的方式实现了对图的系统性遍历，它优先探索深度而非广度，适用于需要深入探索路径的场景。递归实现代码简洁但可能面临栈溢出风险，而迭代实现使用显式栈避免了这个问题。

## 11.2 广度优先搜索

### 广度优先搜索（BFS）基础概念

广度优先搜索（Breadth-First Search，简称BFS）是另一种重要的图遍历算法。与DFS不同，BFS从起始顶点开始，先访问所有距离为1的邻接顶点，再访问所有距离为2的顶点，依此类推，逐层向外扩展。

BFS的核心特点是"先入先出"，这使得它能够找到从起始顶点到其他顶点的最短路径（在无权图中）。BFS通常使用队列数据结构来实现，确保按照层级顺序访问顶点。

BFS的主要应用场景包括：

- 在无权图中寻找最短路径
- 检测图的二分性
- 网络爬虫的页面抓取
- 社交网络中的好友推荐（按距离远近）
- 迷宫求解（寻找最短路径）

### BFS实现方法表格

| 功能名称    | 实现方法                           | 具体说明                       |
| ----------- | ---------------------------------- | ------------------------------ |
| 基础BFS     | bfs(graph, start, visited)         | 使用队列实现标准BFS遍历        |
| 最短路径BFS | bfsShortestPath(graph, start, end) | 记录路径信息，用于寻找最短路径 |
| 层级遍历    | bfsLevelOrder(graph, start)        | 按层级输出遍历结果             |

### 队列实现广度优先搜索

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

// 定义队列结构
typedef struct {
    int items[MAX_VERTICES];
    int front;
    int rear;
} Queue;

// 初始化队列
void initQueue(Queue* queue) {
    // 设置队列前后指针
    queue->front = 0;
    queue->rear = -1;
}

// 判断队列是否为空
bool isEmpty(Queue* queue) {
    // 当rear小于front时表示队列为空
    return queue->rear < queue->front;
}

// 入队操作
void enqueue(Queue* queue, int value) {
    // 检查队列是否已满
    if (queue->rear >= MAX_VERTICES - 1) {
        printf("队列溢出!\n");
        return;
    }
    // 增加rear指针并存储值
    queue->items[++(queue->rear)] = value;
}

// 出队操作
int dequeue(Queue* queue) {
    // 检查队列是否为空
    if (isEmpty(queue)) {
        printf("队列为空!\n");
        return -1;
    }
    // 返回front位置的元素并增加front指针
    return queue->items[(queue->front)++];
}

// 广度优先搜索实现
void bfs(int start, int graph[][MAX_VERTICES], int numVertices) {
    // 创建访问标记数组并初始化为false
    bool visited[MAX_VERTICES] = {false};
    // 创建队列
    Queue queue;
    initQueue(&queue);
    
    // 标记起始顶点为已访问并入队
    visited[start] = true;
    enqueue(&queue, start);
    
    printf("BFS遍历顺序: ");
    
    // 当队列不为空时继续
    while (!isEmpty(&queue)) {
        // 出队一个顶点
        int current = dequeue(&queue);
        // 打印当前顶点
        printf("%d ", current);
        
        // 遍历所有邻接顶点
        for (int i = 0; i < numVertices; i++) {
            // 如果存在边且邻接顶点未被访问
            if (graph[current][i] == 1 && !visited[i]) {
                // 标记为已访问并入队
                visited[i] = true;
                enqueue(&queue, i);
            }
        }
    }
    printf("\n");
}

// 主函数 - 演示BFS
int main() {
    // 创建图的邻接矩阵
    int graph[MAX_VERTICES][MAX_VERTICES] = {0};
    int numVertices = 6;
    
    // 构建图（与DFS示例相同的图结构）
    graph[0][1] = graph[1][0] = 1;  // 边0-1
    graph[0][2] = graph[2][0] = 1;  // 边0-2
    graph[1][3] = graph[3][1] = 1;  // 边1-3
    graph[2][4] = graph[4][2] = 1;  // 边2-4
    graph[3][5] = graph[5][3] = 1;  // 边3-5
    
    // 执行BFS遍历
    bfs(0, graph, numVertices);
    
    return 0;
}
```

### BFS最短路径实现

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

// 路径信息结构
typedef struct {
    int distance;   // 从起始点到当前点的距离
    int parent;     // 当前点的父节点（用于重构路径）
} PathInfo;

// BFS最短路径查找
void bfsShortestPath(int start, int end, int graph[][MAX_VERTICES], int numVertices) {
    // 创建路径信息数组
    PathInfo pathInfo[MAX_VERTICES];
    // 创建访问标记数组
    bool visited[MAX_VERTICES] = {false};
    // 创建队列
    Queue queue;
    initQueue(&queue);
    
    // 初始化所有顶点的路径信息
    for (int i = 0; i < numVertices; i++) {
        pathInfo[i].distance = -1;  // -1表示不可达
        pathInfo[i].parent = -1;    // -1表示无父节点
    }
    
    // 初始化起始顶点
    visited[start] = true;
    pathInfo[start].distance = 0;
    pathInfo[start].parent = -1;
    enqueue(&queue, start);
    
    // BFS主循环
    while (!isEmpty(&queue)) {
        int current = dequeue(&queue);
        
        // 如果找到了目标顶点，提前结束
        if (current == end) {
            break;
        }
        
        // 遍历所有邻接顶点
        for (int i = 0; i < numVertices; i++) {
            if (graph[current][i] == 1 && !visited[i]) {
                visited[i] = true;
                pathInfo[i].distance = pathInfo[current].distance + 1;
                pathInfo[i].parent = current;
                enqueue(&queue, i);
            }
        }
    }
    
    // 输出结果
    if (pathInfo[end].distance == -1) {
        printf("顶点 %d 到顶点 %d 不可达\n", start, end);
    } else {
        printf("从顶点 %d 到顶点 %d 的最短距离: %d\n", 
               start, end, pathInfo[end].distance);
        
        // 重构并打印路径
        printf("最短路径: ");
        int path[MAX_VERTICES];
        int pathIndex = 0;
        int current = end;
        
        // 从终点回溯到起点
        while (current != -1) {
            path[pathIndex++] = current;
            current = pathInfo[current].parent;
        }
        
        // 反向打印路径（从起点到终点）
        for (int i = pathIndex - 1; i >= 0; i--) {
            printf("%d", path[i]);
            if (i > 0) printf(" -> ");
        }
        printf("\n");
    }
}

// 主函数 - 演示BFS最短路径
int main() {
    // 创建图
    int graph[MAX_VERTICES][MAX_VERTICES] = {0};
    int numVertices = 6;
    
    // 构建相同的图结构
    graph[0][1] = graph[1][0] = 1;
    graph[0][2] = graph[2][0] = 1;
    graph[1][3] = graph[3][1] = 1;
    graph[2][4] = graph[4][2] = 1;
    graph[3][5] = graph[5][3] = 1;
    
    // 查找从顶点0到顶点5的最短路径
    bfsShortestPath(0, 5, graph, numVertices);
    
    // 查找从顶点0到顶点4的最短路径
    bfsShortestPath(0, 4, graph, numVertices);
    
    return 0;
}
```

广度优先搜索通过队列实现了按层级的图遍历，保证了在无权图中找到最短路径的最优性。BFS的时间复杂度为O(V+E)，其中V是顶点数，E是边数，空间复杂度主要由队列和访问数组决定，为O(V)。

## 11.3 连通性判断

### 图连通性基础概念

图的连通性是指图中任意两个顶点之间是否存在路径。对于无向图，如果任意两个顶点之间都存在路径，则称该图为连通图；否则称为非连通图。非连通图由多个连通分量组成，每个连通分量都是一个极大连通子图。

对于有向图，连通性的概念更加复杂：

- **弱连通**：忽略边的方向后，对应的无向图是连通的
- **单向连通**：任意两个顶点u和v，要么存在从u到v的路径，要么存在从v到u的路径
- **强连通**：任意两个顶点u和v，既存在从u到v的路径，也存在从v到u的路径

连通性判断在实际应用中非常重要，例如：

- 网络可靠性分析（判断网络是否连通）
- 社交网络分析（识别社区结构）
- 电路设计（确保信号能够到达所有组件）
- 交通网络规划（确保所有地点可达）

### 连通性判断方法表格

| 功能名称     | 实现方法                                            | 具体说明                               |
| ------------ | --------------------------------------------------- | -------------------------------------- |
| 无向图连通性 | isConnectedUndirected(graph, numVertices)           | 使用DFS或BFS判断无向图是否连通         |
| 连通分量计数 | countConnectedComponents(graph, numVertices)        | 统计无向图中连通分量的数量             |
| 强连通分量   | findStronglyConnectedComponents(graph, numVertices) | 使用Kosaraju算法找出有向图的强连通分量 |

### 无向图连通性判断

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

// 使用DFS判断无向图连通性
bool isConnectedUndirected(int graph[][MAX_VERTICES], int numVertices) {
    // 特殊情况：空图或单顶点图
    if (numVertices <= 1) {
        return true;
    }
    
    // 创建访问标记数组
    bool visited[MAX_VERTICES] = {false};
    
    // 从顶点0开始DFS
    dfsHelper(0, graph, visited, numVertices);
    
    // 检查是否所有顶点都被访问
    for (int i = 0; i < numVertices; i++) {
        if (!visited[i]) {
            return false;  // 存在未访问顶点，图不连通
        }
    }
    return true;  // 所有顶点都被访问，图连通
}

// DFS辅助函数
void dfsHelper(int vertex, int graph[][MAX_VERTICES], bool visited[], int numVertices) {
    // 标记当前顶点为已访问
    visited[vertex] = true;
    
    // 遍历所有邻接顶点
    for (int i = 0; i < numVertices; i++) {
        if (graph[vertex][i] == 1 && !visited[i]) {
            dfsHelper(i, graph, visited, numVertices);
        }
    }
}

// 统计连通分量数量
int countConnectedComponents(int graph[][MAX_VERTICES], int numVertices) {
    // 创建访问标记数组
    bool visited[MAX_VERTICES] = {false};
    int components = 0;
    
    // 遍历所有顶点
    for (int i = 0; i < numVertices; i++) {
        if (!visited[i]) {
            // 发现新的连通分量
            components++;
            // 从该顶点开始DFS，标记整个连通分量
            dfsHelper(i, graph, visited, numVertices);
        }
    }
    return components;
}

// 打印连通分量
void printConnectedComponents(int graph[][MAX_VERTICES], int numVertices) {
    bool visited[MAX_VERTICES] = {false};
    int componentId = 1;
    
    printf("连通分量:\n");
    
    for (int i = 0; i < numVertices; i++) {
        if (!visited[i]) {
            printf("连通分量 %d: ", componentId++);
            // 使用栈实现的DFS来打印连通分量
            printComponent(i, graph, visited, numVertices);
            printf("\n");
        }
    }
}

// 打印单个连通分量的辅助函数
void printComponent(int start, int graph[][MAX_VERTICES], bool visited[], int numVertices) {
    // 使用栈避免递归
    int stack[MAX_VERTICES];
    int top = -1;
    
    // 将起始顶点入栈
    stack[++top] = start;
    visited[start] = true;
    
    while (top >= 0) {
        int current = stack[top--];
        printf("%d ", current);
        
        // 将所有未访问的邻接顶点入栈
        for (int i = 0; i < numVertices; i++) {
            if (graph[current][i] == 1 && !visited[i]) {
                visited[i] = true;
                stack[++top] = i;
            }
        }
    }
}

// 主函数 - 演示连通性判断
int main() {
    // 测试用例1：连通图
    int graph1[MAX_VERTICES][MAX_VERTICES] = {0};
    int numVertices1 = 5;
    
    // 构建连通图
    graph1[0][1] = graph1[1][0] = 1;
    graph1[1][2] = graph1[2][1] = 1;
    graph1[2][3] = graph1[3][2] = 1;
    graph1[3][4] = graph1[4][3] = 1;
    
    printf("测试用例1 - 连通图:\n");
    printf("是否连通: %s\n", isConnectedUndirected(graph1, numVertices1) ? "是" : "否");
    printf("连通分量数量: %d\n", countConnectedComponents(graph1, numVertices1));
    printConnectedComponents(graph1, numVertices1);
    
    printf("\n");
    
    // 测试用例2：非连通图
    int graph2[MAX_VERTICES][MAX_VERTICES] = {0};
    int numVertices2 = 6;
    
    // 构建非连通图（两个连通分量）
    graph2[0][1] = graph2[1][0] = 1;
    graph2[1][2] = graph2[2][1] = 1;
    graph2[3][4] = graph2[4][3] = 1;
    graph2[4][5] = graph2[5][4] = 1;
    // 顶点0,1,2构成一个连通分量，顶点3,4,5构成另一个连通分量
    
    printf("测试用例2 - 非连通图:\n");
    printf("是否连通: %s\n", isConnectedUndirected(graph2, numVertices2) ? "是" : "否");
    printf("连通分量数量: %d\n", countConnectedComponents(graph2, numVertices2));
    printConnectedComponents(graph2, numVertices2);
    
    return 0;
}
```

### 有向图强连通分量（Kosaraju算法）

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

// Kosaraju算法实现强连通分量
void findStronglyConnectedComponents(int graph[][MAX_VERTICES], int numVertices) {
    // 步骤1：第一次DFS，记录完成时间
    bool visited[MAX_VERTICES] = {false};
    int stack[MAX_VERTICES];
    int top = -1;
    
    // 对所有顶点进行DFS
    for (int i = 0; i < numVertices; i++) {
        if (!visited[i]) {
            fillOrder(i, graph, visited, stack, &top, numVertices);
        }
    }
    
    // 步骤2：转置图（反转所有边的方向）
    int transpose[MAX_VERTICES][MAX_VERTICES] = {0};
    for (int i = 0; i < numVertices; i++) {
        for (int j = 0; j < numVertices; j++) {
            if (graph[i][j] == 1) {
                transpose[j][i] = 1;
            }
        }
    }
    
    // 步骤3：按完成时间的逆序对转置图进行DFS
    for (int i = 0; i < numVertices; i++) {
        visited[i] = false;
    }
    
    printf("强连通分量:\n");
    int componentCount = 0;
    
    while (top >= 0) {
        int v = stack[top--];
        if (!visited[v]) {
            printf("SCC %d: ", ++componentCount);
            dfsTranspose(v, transpose, visited, numVertices);
            printf("\n");
        }
    }
}

// 第一次DFS，按完成时间将顶点压入栈
void fillOrder(int v, int graph[][MAX_VERTICES], bool visited[], 
               int stack[], int* top, int numVertices) {
    visited[v] = true;
    
    for (int i = 0; i < numVertices; i++) {
        if (graph[v][i] == 1 && !visited[i]) {
            fillOrder(i, graph, visited, stack, top, numVertices);
        }
    }
    
    // 完成DFS后将顶点压入栈
    stack[++(*top)] = v;
}

// 在转置图上进行DFS
void dfsTranspose(int v, int transpose[][MAX_VERTICES], bool visited[], int numVertices) {
    visited[v] = true;
    printf("%d ", v);
    
    for (int i = 0; i < numVertices; i++) {
        if (transpose[v][i] == 1 && !visited[i]) {
            dfsTranspose(i, transpose, visited, numVertices);
        }
    }
}

// 主函数 - 演示强连通分量
int main() {
    // 创建有向图
    int graph[MAX_VERTICES][MAX_VERTICES] = {0};
    int numVertices = 5;
    
    // 构建有向图
    graph[0][1] = 1;
    graph[1][2] = 1;
    graph[2][0] = 1;  // 0,1,2形成一个强连通分量
    graph[1][3] = 1;
    graph[3][4] = 1;  // 3,4各自为单独的强连通分量
    
    printf("有向图的强连通分量:\n");
    findStronglyConnectedComponents(graph, numVertices);
    
    return 0;
}
```

连通性判断是图论中的基础问题，通过DFS或BFS可以有效解决无向图的连通性问题，而Kosaraju算法等专门算法则用于处理有向图的强连通分量识别。这些算法在实际应用中有着广泛用途，从网络分析到社交网络挖掘都离不开连通性分析。
