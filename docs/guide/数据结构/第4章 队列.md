### 4.1 队列的基本概念

队列（Queue）是一种特殊的线性表，它只允许在表的一端进行插入操作，在另一端进行删除操作。这种结构遵循“先进先出”（First In First Out, FIFO）的原则，就像排队买票一样，先来的人先得到服务。

| 技术点        | 版本/环境要求 | 使用场景                             |
| ------------- | ------------- | ------------------------------------ |
| C语言队列基础 | C99标准或以上 | 任务调度、缓冲区管理、广度优先搜索等 |

队列的基本操作包括：

- 入队（Enqueue）：在队尾添加元素
- 出队（Dequeue）：从队头移除元素
- 获取队头元素（Front）：查看但不移除队头元素
- 判断队列是否为空（IsEmpty）
- 获取队列大小（Size）

队列在实际应用中非常广泛，比如操作系统中的进程调度、打印任务队列、网络数据包处理等。理解队列的基本概念是学习更复杂数据结构的基础。

### 4.2 顺序队列实现

顺序队列是使用数组来实现的队列结构。我们需要维护两个指针：front指向队头元素，rear指向队尾元素的下一个位置。

| 功能名称   | 实例调用方法                            | 具体功能与注意事项                                  |
| ---------- | --------------------------------------- | --------------------------------------------------- |
| 初始化队列 | `initQueue(Queue* q, int size)`         | 分配内存并初始化front和rear为0                      |
| 入队操作   | `enqueue(Queue* q, int data)`           | 检查队列是否已满，未满则在rear位置插入元素          |
| 出队操作   | `dequeue(Queue* q)`                     | 检查队列是否为空，非空则返回front位置元素并移动指针 |
| 获取队头   | `front(Queue* q)`                       | 返回队头元素但不移动指针，需检查队列是否为空        |
| 队列状态   | `isEmpty(Queue* q)`, `isFull(Queue* q)` | 判断队列是否为空或已满                              |

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

// 定义队列结构
typedef struct {
    int* data;      // 存储队列元素的数组
    int front;      // 队头指针
    int rear;       // 队尾指针
    int capacity;   // 队列容量
} Queue;

// 初始化队列
Queue* initQueue(int size) {
    // 分配队列结构内存
    Queue* q = (Queue*)malloc(sizeof(Queue));
    if (!q) {
        printf("内存分配失败\n");
        return NULL;
    }
    // 分配数据数组内存
    q->data = (int*)malloc(size * sizeof(int));
    if (!q->data) {
        free(q);
        printf("内存分配失败\n");
        return NULL;
    }
    // 初始化指针和容量
    q->front = 0;
    q->rear = 0;
    q->capacity = size;
    return q;
}

// 检查队列是否为空
bool isEmpty(Queue* q) {
    // 当front等于rear时队列为空
    return q->front == q->rear;
}

// 检查队列是否已满
bool isFull(Queue* q) {
    // 当rear达到容量上限时队列已满
    return q->rear == q->capacity;
}

// 入队操作
bool enqueue(Queue* q, int data) {
    // 先检查队列是否已满
    if (isFull(q)) {
        printf("队列已满，无法入队\n");
        return false;
    }
    // 在rear位置插入元素
    q->data[q->rear] = data;
    // rear指针后移
    q->rear++;
    return true;
}

// 出队操作
bool dequeue(Queue* q, int* data) {
    // 先检查队列是否为空
    if (isEmpty(q)) {
        printf("队列为空，无法出队\n");
        return false;
    }
    // 获取front位置的元素
    *data = q->data[q->front];
    // front指针后移
    q->front++;
    return true;
}

// 获取队头元素
bool getFront(Queue* q, int* data) {
    // 检查队列是否为空
    if (isEmpty(q)) {
        printf("队列为空\n");
        return false;
    }
    // 返回队头元素
    *data = q->data[q->front];
    return true;
}

// 销毁队列
void destroyQueue(Queue* q) {
    // 释放数据数组内存
    free(q->data);
    // 释放队列结构内存
    free(q);
}

// 测试函数
int main() {
    // 创建容量为5的队列
    Queue* q = initQueue(5);
    if (!q) return -1;
    
    // 测试入队
    printf("入队操作:\n");
    for (int i = 1; i <= 3; i++) {
        if (enqueue(q, i)) {
            printf("成功入队: %d\n", i);
        }
    }
    
    // 测试获取队头
    int frontData;
    if (getFront(q, &frontData)) {
        printf("队头元素: %d\n", frontData);
    }
    
    // 测试出队
    printf("出队操作:\n");
    int dequeued;
    while (!isEmpty(q)) {
        if (dequeue(q, &dequeued)) {
            printf("成功出队: %d\n", dequeued);
        }
    }
    
    // 销毁队列
    destroyQueue(q);
    return 0;
}
```

**注意事项：**

1. 顺序队列存在"假溢出"问题：当rear达到数组末尾时，即使前面有空闲空间也无法继续入队
2. 内存管理很重要，使用完队列后必须释放分配的内存
3. 所有操作都需要进行边界检查，避免访问无效内存
4. 出队操作只是移动front指针，并没有真正清除数据，但这不影响队列的正确性

顺序队列实现简单直观，但由于假溢出问题，实际应用中更多使用循环队列。

### 4.3 循环队列实现

循环队列解决了顺序队列的"假溢出"问题，通过将数组视为环形结构，当rear到达数组末尾时可以回到开头继续使用前面的空闲空间。

| 功能名称       | 实例调用方法                                                 | 具体功能与注意事项                            |
| -------------- | ------------------------------------------------------------ | --------------------------------------------- |
| 初始化循环队列 | `initCircularQueue(CircularQueue* cq, int size)`             | 创建大小为size+1的数组（牺牲一个空间）        |
| 入队操作       | `enqueueCircular(CircularQueue* cq, int data)`               | 使用(rear+1)%capacity判断是否已满             |
| 出队操作       | `dequeueCircular(CircularQueue* cq)`                         | 使用front = (front+1)%capacity移动指针        |
| 队列状态判断   | `isEmptyCircular(CircularQueue* cq)`, `isFullCircular(CircularQueue* cq)` | 空：front==rear；满：(rear+1)%capacity==front |

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

// 定义循环队列结构
typedef struct {
    int* data;      // 存储队列元素的数组
    int front;      // 队头指针
    int rear;       // 队尾指针
    int capacity;   // 队列容量（实际数组大小为capacity+1）
} CircularQueue;

// 初始化循环队列
CircularQueue* initCircularQueue(int size) {
    // 分配循环队列结构内存
    CircularQueue* cq = (CircularQueue*)malloc(sizeof(CircularQueue));
    if (!cq) {
        printf("内存分配失败\n");
        return NULL;
    }
    // 分配数据数组内存（多分配一个空间用于区分空和满）
    cq->data = (int*)malloc((size + 1) * sizeof(int));
    if (!cq->data) {
        free(cq);
        printf("内存分配失败\n");
        return NULL;
    }
    // 初始化指针和容量
    cq->front = 0;
    cq->rear = 0;
    cq->capacity = size + 1;  // 实际容量比用户指定的大1
    return cq;
}

// 检查循环队列是否为空
bool isEmptyCircular(CircularQueue* cq) {
    // 当front等于rear时队列为空
    return cq->front == cq->rear;
}

// 检查循环队列是否已满
bool isFullCircular(CircularQueue* cq) {
    // 当(rear+1)%capacity等于front时队列已满
    return (cq->rear + 1) % cq->capacity == cq->front;
}

// 入队操作
bool enqueueCircular(CircularQueue* cq, int data) {
    // 先检查队列是否已满
    if (isFullCircular(cq)) {
        printf("循环队列已满，无法入队\n");
        return false;
    }
    // 在rear位置插入元素
    cq->data[cq->rear] = data;
    // rear指针循环后移
    cq->rear = (cq->rear + 1) % cq->capacity;
    return true;
}

// 出队操作
bool dequeueCircular(CircularQueue* cq, int* data) {
    // 先检查队列是否为空
    if (isEmptyCircular(cq)) {
        printf("循环队列为空，无法出队\n");
        return false;
    }
    // 获取front位置的元素
    *data = cq->data[cq->front];
    // front指针循环后移
    cq->front = (cq->front + 1) % cq->capacity;
    return true;
}

// 获取队头元素
bool getFrontCircular(CircularQueue* cq, int* data) {
    // 检查队列是否为空
    if (isEmptyCircular(cq)) {
        printf("循环队列为空\n");
        return false;
    }
    // 返回队头元素
    *data = cq->data[cq->front];
    return true;
}

// 获取队列大小
int sizeCircular(CircularQueue* cq) {
    // 计算队列中元素个数
    return (cq->rear - cq->front + cq->capacity) % cq->capacity;
}

// 销毁循环队列
void destroyCircularQueue(CircularQueue* cq) {
    // 释放数据数组内存
    free(cq->data);
    // 释放队列结构内存
    free(cq);
}

// 测试函数
int main() {
    // 创建容量为5的循环队列
    CircularQueue* cq = initCircularQueue(5);
    if (!cq) return -1;
    
    // 测试入队
    printf("入队操作:\n");
    for (int i = 1; i <= 5; i++) {
        if (enqueueCircular(cq, i)) {
            printf("成功入队: %d\n", i);
        }
    }
    
    // 尝试再次入队（应该失败）
    enqueueCircular(cq, 6);
    
    // 测试出队
    printf("\n出队操作:\n");
    int dequeued;
    for (int i = 0; i < 2; i++) {
        if (dequeueCircular(cq, &dequeued)) {
            printf("成功出队: %d\n", dequeued);
        }
    }
    
    // 再次入队（利用循环特性）
    printf("\n再次入队:\n");
    if (enqueueCircular(cq, 6)) {
        printf("成功入队: %d\n", 6);
    }
    if (enqueueCircular(cq, 7)) {
        printf("成功入队: %d\n", 7);
    }
    
    // 显示当前队列大小
    printf("\n当前队列大小: %d\n", sizeCircular(cq));
    
    // 继续出队直到空
    printf("\n清空队列:\n");
    while (!isEmptyCircular(cq)) {
        if (dequeueCircular(cq, &dequeued)) {
            printf("成功出队: %d\n", dequeued);
        }
    }
    
    // 销毁队列
    destroyCircularQueue(cq);
    return 0;
}
```

**注意事项：**

1. 循环队列牺牲一个存储空间来区分空和满状态
2. 指针移动必须使用取模运算：`(pointer + 1) % capacity`
3. 计算队列大小时要考虑循环情况：`(rear - front + capacity) % capacity`
4. 虽然解决了假溢出问题，但仍然有固定容量限制
5. 循环队列的空间利用率比顺序队列高，但实现稍微复杂一些

循环队列通过巧妙的数学运算解决了顺序队列的空间浪费问题，是实际应用中常用的队列实现方式。

### 4.4 链队列实现

链队列使用链表来实现队列，克服了顺序队列和循环队列固定容量的限制，可以动态分配内存。

| 功能名称     | 实例调用方法                               | 具体功能与注意事项                  |
| ------------ | ------------------------------------------ | ----------------------------------- |
| 初始化链队列 | `initLinkedQueue()`                        | 创建头节点，front和rear都指向头节点 |
| 入队操作     | `enqueueLinked(LinkedQueue* lq, int data)` | 在rear后插入新节点，更新rear指针    |
| 出队操作     | `dequeueLinked(LinkedQueue* lq)`           | 删除front后的节点，更新front指针    |
| 队列状态判断 | `isEmptyLinked(LinkedQueue* lq)`           | 检查front是否等于rear               |

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

// 定义队列节点结构
typedef struct Node {
    int data;           // 节点数据
    struct Node* next;  // 指向下一个节点的指针
} Node;

// 定义链队列结构
typedef struct {
    Node* front;  // 队头指针（指向头节点）
    Node* rear;   // 队尾指针（指向最后一个节点）
} LinkedQueue;

// 初始化链队列
LinkedQueue* initLinkedQueue() {
    // 分配链队列结构内存
    LinkedQueue* lq = (LinkedQueue*)malloc(sizeof(LinkedQueue));
    if (!lq) {
        printf("内存分配失败\n");
        return NULL;
    }
    // 创建头节点（不存储实际数据）
    lq->front = (Node*)malloc(sizeof(Node));
    if (!lq->front) {
        free(lq);
        printf("内存分配失败\n");
        return NULL;
    }
    // 初始化头节点
    lq->front->next = NULL;
    // rear指向头节点（初始时队列为空）
    lq->rear = lq->front;
    return lq;
}

// 检查链队列是否为空
bool isEmptyLinked(LinkedQueue* lq) {
    // 当front等于rear时队列为空
    return lq->front == lq->rear;
}

// 入队操作
bool enqueueLinked(LinkedQueue* lq, int data) {
    // 创建新节点
    Node* newNode = (Node*)malloc(sizeof(Node));
    if (!newNode) {
        printf("内存分配失败\n");
        return false;
    }
    // 设置新节点数据
    newNode->data = data;
    newNode->next = NULL;
    // 将新节点链接到队尾
    lq->rear->next = newNode;
    // 更新rear指针
    lq->rear = newNode;
    return true;
}

// 出队操作
bool dequeueLinked(LinkedQueue* lq, int* data) {
    // 检查队列是否为空
    if (isEmptyLinked(lq)) {
        printf("链队列为空，无法出队\n");
        return false;
    }
    // 获取要删除的节点（front后的第一个节点）
    Node* temp = lq->front->next;
    // 获取节点数据
    *data = temp->data;
    // 更新front指针
    lq->front->next = temp->next;
    // 如果删除的是最后一个节点，更新rear指针
    if (lq->rear == temp) {
        lq->rear = lq->front;
    }
    // 释放被删除节点的内存
    free(temp);
    return true;
}

// 获取队头元素
bool getFrontLinked(LinkedQueue* lq, int* data) {
    // 检查队列是否为空
    if (isEmptyLinked(lq)) {
        printf("链队列为空\n");
        return false;
    }
    // 返回队头元素（front后的第一个节点的数据）
    *data = lq->front->next->data;
    return true;
}

// 销毁链队列
void destroyLinkedQueue(LinkedQueue* lq) {
    // 从头节点开始逐个释放节点
    Node* current = lq->front;
    while (current != NULL) {
        Node* temp = current;
        current = current->next;
        free(temp);
    }
    // 释放队列结构内存
    free(lq);
}

// 测试函数
int main() {
    // 初始化链队列
    LinkedQueue* lq = initLinkedQueue();
    if (!lq) return -1;
    
    // 测试入队
    printf("入队操作:\n");
    for (int i = 1; i <= 5; i++) {
        if (enqueueLinked(lq, i)) {
            printf("成功入队: %d\n", i);
        }
    }
    
    // 测试获取队头
    int frontData;
    if (getFrontLinked(lq, &frontData)) {
        printf("队头元素: %d\n", frontData);
    }
    
    // 测试出队
    printf("\n出队操作:\n");
    int dequeued;
    for (int i = 0; i < 3; i++) {
        if (dequeueLinked(lq, &dequeued)) {
            printf("成功出队: %d\n", dequeued);
        }
    }
    
    // 再次入队
    printf("\n再次入队:\n");
    if (enqueueLinked(lq, 6)) {
        printf("成功入队: %d\n", 6);
    }
    
    // 清空队列
    printf("\n清空队列:\n");
    while (!isEmptyLinked(lq)) {
        if (dequeueLinked(lq, &dequeued)) {
            printf("成功出队: %d\n", dequeued);
        }
    }
    
    // 销毁队列
    destroyLinkedQueue(lq);
    return 0;
}
```

**注意事项：**

1. 链队列使用带头节点的单链表实现，简化了边界条件处理
2. 入队操作时间复杂度为O(1)，因为有rear指针直接指向队尾
3. 出队操作时间复杂度为O(1)，因为有front指针直接指向队头
4. 需要仔细管理内存，避免内存泄漏
5. 链队列没有容量限制，但每个节点需要额外的指针空间开销
6. 在删除最后一个元素时，需要特别注意更新rear指针

链队列结合了链表的灵活性和队列的操作特性，适合需要动态调整大小的场景。

### 4.5 双端队列

双端队列（Double-ended Queue，简称Deque）是一种特殊的队列，允许在两端都进行插入和删除操作。它结合了栈和队列的特性，既可以像队列一样FIFO，也可以像栈一样LIFO。

| 功能名称     | 实例调用方法                                | 具体功能与注意事项         |
| ------------ | ------------------------------------------- | -------------------------- |
| 前端入队     | `addFront(Deque* dq, int data)`             | 在队列前端插入元素         |
| 后端入队     | `addRear(Deque* dq, int data)`              | 在队列后端插入元素         |
| 前端出队     | `removeFront(Deque* dq)`                    | 从队列前端删除元素         |
| 后端出队     | `removeRear(Deque* dq)`                     | 从队列后端删除元素         |
| 获取端点元素 | `getFront(Deque* dq)`, `getRear(Deque* dq)` | 获取但不删除前端或后端元素 |

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

// 定义双端队列节点结构
typedef struct DequeNode {
    int data;                   // 节点数据
    struct DequeNode* prev;     // 指向前一个节点的指针
    struct DequeNode* next;     // 指向后一个节点的指针
} DequeNode;

// 定义双端队列结构
typedef struct {
    DequeNode* front;   // 指向队列前端的指针
    DequeNode* rear;    // 指向队列后端的指针
    int size;           // 队列大小
} Deque;

// 初始化双端队列
Deque* initDeque() {
    // 分配双端队列结构内存
    Deque* dq = (Deque*)malloc(sizeof(Deque));
    if (!dq) {
        printf("内存分配失败\n");
        return NULL;
    }
    // 初始化指针和大小
    dq->front = NULL;
    dq->rear = NULL;
    dq->size = 0;
    return dq;
}

// 检查双端队列是否为空
bool isEmptyDeque(Deque* dq) {
    // 当front为NULL时队列为空
    return dq->front == NULL;
}

// 前端入队
bool addFront(Deque* dq, int data) {
    // 创建新节点
    DequeNode* newNode = (DequeNode*)malloc(sizeof(DequeNode));
    if (!newNode) {
        printf("内存分配失败\n");
        return false;
    }
    // 设置新节点数据
    newNode->data = data;
    newNode->prev = NULL;
    // 如果队列为空
    if (isEmptyDeque(dq)) {
        newNode->next = NULL;
        dq->front = dq->rear = newNode;
    } else {
        // 链接到现有队列前端
        newNode->next = dq->front;
        dq->front->prev = newNode;
        dq->front = newNode;
    }
    // 增加队列大小
    dq->size++;
    return true;
}

// 后端入队
bool addRear(Deque* dq, int data) {
    // 创建新节点
    DequeNode* newNode = (DequeNode*)malloc(sizeof(DequeNode));
    if (!newNode) {
        printf("内存分配失败\n");
        return false;
    }
    // 设置新节点数据
    newNode->data = data;
    newNode->next = NULL;
    // 如果队列为空
    if (isEmptyDeque(dq)) {
        newNode->prev = NULL;
        dq->front = dq->rear = newNode;
    } else {
        // 链接到现有队列后端
        newNode->prev = dq->rear;
        dq->rear->next = newNode;
        dq->rear = newNode;
    }
    // 增加队列大小
    dq->size++;
    return true;
}

// 前端出队
bool removeFront(Deque* dq, int* data) {
    // 检查队列是否为空
    if (isEmptyDeque(dq)) {
        printf("双端队列为空，无法从前端出队\n");
        return false;
    }
    // 获取要删除的节点
    DequeNode* temp = dq->front;
    // 获取节点数据
    *data = temp->data;
    // 如果只有一个节点
    if (dq->front == dq->rear) {
        dq->front = dq->rear = NULL;
    } else {
        // 更新front指针
        dq->front = dq->front->next;
        dq->front->prev = NULL;
    }
    // 释放被删除节点的内存
    free(temp);
    // 减少队列大小
    dq->size--;
    return true;
}

// 后端出队
bool removeRear(Deque* dq, int* data) {
    // 检查队列是否为空
    if (isEmptyDeque(dq)) {
        printf("双端队列为空，无法从后端出队\n");
        return false;
    }
    // 获取要删除的节点
    DequeNode* temp = dq->rear;
    // 获取节点数据
    *data = temp->data;
    // 如果只有一个节点
    if (dq->front == dq->rear) {
        dq->front = dq->rear = NULL;
    } else {
        // 更新rear指针
        dq->rear = dq->rear->prev;
        dq->rear->next = NULL;
    }
    // 释放被删除节点的内存
    free(temp);
    // 减少队列大小
    dq->size--;
    return true;
}

// 获取前端元素
bool getFront(Deque* dq, int* data) {
    // 检查队列是否为空
    if (isEmptyDeque(dq)) {
        printf("双端队列为空\n");
        return false;
    }
    // 返回前端元素
    *data = dq->front->data;
    return true;
}

// 获取后端元素
bool getRear(Deque* dq, int* data) {
    // 检查队列是否为空
    if (isEmptyDeque(dq)) {
        printf("双端队列为空\n");
        return false;
    }
    // 返回后端元素
    *data = dq->rear->data;
    return true;
}

// 获取队列大小
int getSize(Deque* dq) {
    return dq->size;
}

// 销毁双端队列
void destroyDeque(Deque* dq) {
    // 从front开始逐个释放节点
    while (!isEmptyDeque(dq)) {
        int dummy;
        removeFront(dq, &dummy);
    }
    // 释放队列结构内存
    free(dq);
}

// 测试函数
int main() {
    // 初始化双端队列
    Deque* dq = initDeque();
    if (!dq) return -1;
    
    // 测试前端入队
    printf("前端入队:\n");
    for (int i = 1; i <= 3; i++) {
        if (addFront(dq, i)) {
            printf("前端入队: %d\n", i);
        }
    }
    
    // 测试后端入队
    printf("\n后端入队:\n");
    for (int i = 4; i <= 6; i++) {
        if (addRear(dq, i)) {
            printf("后端入队: %d\n", i);
        }
    }
    
    // 显示当前队列大小
    printf("\n当前队列大小: %d\n", getSize(dq));
    
    // 获取端点元素
    int frontData, rearData;
    if (getFront(dq, &frontData)) {
        printf("前端元素: %d\n", frontData);
    }
    if (getRear(dq, &rearData)) {
        printf("后端元素: %d\n", rearData);
    }
    
    // 测试前端出队
    printf("\n前端出队:\n");
    int dequeued;
    for (int i = 0; i < 2; i++) {
        if (removeFront(dq, &dequeued)) {
            printf("前端出队: %d\n", dequeued);
        }
    }
    
    // 测试后端出队
    printf("\n后端出队:\n");
    for (int i = 0; i < 2; i++) {
        if (removeRear(dq, &dequeued)) {
            printf("后端出队: %d\n", dequeued);
        }
    }
    
    // 显示剩余元素
    printf("\n剩余元素:\n");
    while (!isEmptyDeque(dq)) {
        if (removeFront(dq, &dequeued)) {
            printf("出队: %d\n", dequeued);
        }
    }
    
    // 销毁队列
    destroyDeque(dq);
    return 0;

```

**注意事项：**

1. 双端队列使用双向链表实现，支持两端的高效操作
2. 每个节点需要两个指针（prev和next），空间开销比单链表大
3. 所有操作的时间复杂度都是O(1)
4. 需要特别注意边界条件，特别是只有一个节点时的处理
5. 双端队列可以模拟栈（只在一端操作）和普通队列（一端入队，另一端出队）
6. 实际应用包括滑动窗口最大值、回文检查等算法问题

双端队列提供了最大的操作灵活性，虽然实现相对复杂，但在需要两端操作的场景下非常有用。
