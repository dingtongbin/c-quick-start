### 12.1 最小生成树

最小生成树（Minimum Spanning Tree, MST）是图论中的一个重要概念，主要用于解决带权无向连通图中如何用最小的总权重连接所有顶点的问题。在实际应用中，比如网络布线、电路设计等领域，最小生成树能帮助我们以最低成本完成连接任务。

| 功能名称       | 实例调用方法          | 具体功能、注意事项、必需参数/可选参数              |
| -------------- | --------------------- | -------------------------------------------------- |
| 判断图是否连通 | is_connected(graph)   | 检查图是否为连通图，必需参数：图的邻接表或邻接矩阵 |
| 获取边的权重   | get_edge_weight(edge) | 返回指定边的权重值，必需参数：边的两个顶点         |

```c
#include <stdio.h>
#include <stdlib.h>
#include <limits.h>

// 定义图的最大顶点数
#define MAX_VERTICES 100

// 边结构体
typedef struct {
    int src;    // 起始顶点
    int dest;   // 目标顶点
    int weight; // 边的权重
} Edge;

// 图结构体
typedef struct {
    int vertices; // 顶点数量
    int edges;    // 边数量
    Edge* edge_list; // 边列表
} Graph;

// 创建图
Graph* create_graph(int vertices, int edges) {
    // 分配图内存
    Graph* graph = (Graph*)malloc(sizeof(Graph));
    // 设置顶点数量
    graph->vertices = vertices;
    // 设置边数量
    graph->edges = edges;
    // 分配边列表内存
    graph->edge_list = (Edge*)malloc(edges * sizeof(Edge));
    // 返回创建的图
    return graph;
}

// 查找顶点的根节点（用于检测环）
int find_parent(int parent[], int i) {
    // 如果当前顶点是自己的父节点，则返回自身
    if (parent[i] == -1)
        return i;
    // 递归查找根节点
    return find_parent(parent, parent[i]);
}

// 合并两个集合（用于Kruskal算法）
void union_sets(int parent[], int x, int y) {
    // 找到x的根节点
    int x_root = find_parent(parent, x);
    // 找到y的根节点
    int y_root = find_parent(parent, y);
    // 将y的根节点设置为x的根节点的子节点
    parent[x_root] = y_root;
}

// 比较函数，用于qsort排序边
int compare_edges(const void* a, const void* b) {
    // 强制转换为Edge指针
    Edge* edge_a = (Edge*)a;
    Edge* edge_b = (Edge*)b;
    // 按权重升序排列
    return edge_a->weight - edge_b->weight;
}

// 检测图中是否存在环
int has_cycle(Graph* graph) {
    // 初始化父数组，-1表示每个顶点初始时都是独立的集合
    int* parent = (int*)malloc(graph->vertices * sizeof(int));
    for (int i = 0; i < graph->vertices; i++) {
        parent[i] = -1;
    }
    
    // 遍历所有边
    for (int i = 0; i < graph->edges; i++) {
        // 获取当前边的起始和目标顶点
        int x = find_parent(parent, graph->edge_list[i].src);
        int y = find_parent(parent, graph->edge_list[i].dest);
        
        // 如果两个顶点属于同一集合，则存在环
        if (x == y) {
            free(parent);
            return 1;
        }
        // 合并两个集合
        union_sets(parent, x, y);
    }
    free(parent);
    return 0;
}

int main() {
    // 创建一个包含4个顶点和5条边的图
    Graph* graph = create_graph(4, 5);
    
    // 添加边（顶点编号从0开始）
    graph->edge_list[0].src = 0;
    graph->edge_list[0].dest = 1;
    graph->edge_list[0].weight = 10;
    
    graph->edge_list[1].src = 0;
    graph->edge_list[1].dest = 2;
    graph->edge_list[1].weight = 6;
    
    graph->edge_list[2].src = 0;
    graph->edge_list[2].dest = 3;
    graph->edge_list[2].weight = 5;
    
    graph->edge_list[3].src = 1;
    graph->edge_list[3].dest = 3;
    graph->edge_list[3].weight = 15;
    
    graph->edge_list[4].src = 2;
    graph->edge_list[4].dest = 3;
    graph->edge_list[4].weight = 4;
    
    // 检查图中是否存在环
    if (has_cycle(graph)) {
        printf("图中存在环\n");
    } else {
        printf("图中不存在环\n");
    }
    
    // 释放内存
    free(graph->edge_list);
    free(graph);
    
    return 0;
}
```

在实现最小生成树算法时，需要特别注意图必须是连通的，否则无法形成生成树。同时，要确保正确处理边的权重比较和集合合并操作，避免出现逻辑错误。

### 12.2 Prim算法

Prim算法是一种贪心算法，用于求解最小生成树问题。它的基本思想是从任意一个顶点开始，逐步扩展已选顶点集合，每次选择连接已选顶点和未选顶点之间权重最小的边，直到所有顶点都被包含在生成树中。

| 功能名称         | 实例调用方法                         | 具体功能、注意事项、必需参数/可选参数                        |
| ---------------- | ------------------------------------ | ------------------------------------------------------------ |
| 初始化距离数组   | init_key_array(key, vertices)        | 将所有顶点的距离初始化为无穷大，必需参数：距离数组和顶点数量 |
| 查找最小键值顶点 | find_min_key(key, mst_set, vertices) | 在未包含在MST中的顶点中找到键值最小的顶点，必需参数：键值数组、MST集合标记数组、顶点数量 |

```c
#include <stdio.h>
#include <limits.h>
#include <stdbool.h>

// 定义图的最大顶点数
#define V 5

// 获取最小键值的顶点
int min_key(int key[], bool mst_set[]) {
    // 初始化最小值为最大整数
    int min = INT_MAX, min_index;
    
    // 遍历所有顶点
    for (int v = 0; v < V; v++) {
        // 如果顶点不在MST中且键值小于当前最小值
        if (mst_set[v] == false && key[v] < min) {
            // 更新最小值和索引
            min = key[v];
            min_index = v;
        }
    }
    // 返回最小键值顶点的索引
    return min_index;
}

// 打印构造的MST
void print_mst(int parent[], int graph[V][V]) {
    printf("边 \t权重\n");
    // 遍历所有顶点（除了根节点）
    for (int i = 1; i < V; i++) {
        // 打印父节点到当前节点的边及其权重
        printf("%d - %d \t%d\n", parent[i], i, graph[i][parent[i]]);
    }
}

// Prim算法主函数
void prim_mst(int graph[V][V]) {
    // 存储MST中每个顶点的父节点
    int parent[V];
    // 存储每个顶点到MST的最小权重
    int key[V];
    // 标记顶点是否已包含在MST中
    bool mst_set[V];
    
    // 初始化所有键值为无穷大，MST集合为false
    for (int i = 0; i < V; i++) {
        key[i] = INT_MAX;
        mst_set[i] = false;
    }
    
    // 从顶点0开始，设置其键值为0
    key[0] = 0;
    // 根节点没有父节点
    parent[0] = -1;
    
    // 构建MST，需要V-1条边
    for (int count = 0; count < V - 1; count++) {
        // 选择键值最小的顶点
        int u = min_key(key, mst_set);
        // 将该顶点加入MST
        mst_set[u] = true;
        
        // 更新相邻顶点的键值
        for (int v = 0; v < V; v++) {
            // 如果存在边且v不在MST中且新权重更小
            if (graph[u][v] && mst_set[v] == false && graph[u][v] < key[v]) {
                // 更新父节点和键值
                parent[v] = u;
                key[v] = graph[u][v];
            }
        }
    }
    
    // 打印结果
    print_mst(parent, graph);
}

int main() {
    // 定义邻接矩阵表示的图
    int graph[V][V] = {
        {0, 2, 0, 6, 0},
        {2, 0, 3, 8, 5},
        {0, 3, 0, 0, 7},
        {6, 8, 0, 0, 9},
        {0, 5, 7, 9, 0}
    };
    
    // 调用Prim算法
    prim_mst(graph);
    
    return 0;
}
```

Prim算法的时间复杂度为O(V²)，其中V是顶点数量。如果使用优先队列优化，可以将时间复杂度降低到O(E log V)，其中E是边的数量。该算法适用于稠密图，即边数接近顶点数平方的情况。

### 12.3 Kruskal算法

Kruskal算法是另一种求解最小生成树的经典算法，它采用贪心策略，按照边的权重从小到大依次选择边，只要不形成环就将其加入生成树中。该算法的核心在于使用并查集数据结构来高效检测环的存在。

| 功能名称   | 实例调用方法                 | 具体功能、注意事项、必需参数/可选参数                 |
| ---------- | ---------------------------- | ----------------------------------------------------- |
| 边排序     | sort_edges(edges, num_edges) | 按权重对边进行升序排序，必需参数：边数组和边数量      |
| 并查集查找 | find(subset, i)              | 查找元素i所在集合的根，必需参数：并查集数组和元素索引 |

```c
#include <stdio.h>
#include <stdlib.h>

// 定义边结构体
typedef struct {
    int src, dest, weight;
} Edge;

// 定义图结构体
typedef struct {
    int V, E;
    Edge* edge;
} Graph;

// 创建图
Graph* create_graph(int V, int E) {
    // 分配图内存
    Graph* graph = (Graph*) malloc(sizeof(Graph));
    // 设置顶点和边数量
    graph->V = V;
    graph->E = E;
    // 分配边数组内存
    graph->edge = (Edge*) malloc(E * sizeof(Edge));
    // 返回图指针
    return graph;
}

// 并查集结构体
typedef struct {
    int parent;
    int rank;
} Subset;

// 查找操作（带路径压缩）
int find(Subset subsets[], int i) {
    // 如果i不是根节点
    if (subsets[i].parent != i)
        // 递归查找并压缩路径
        subsets[i].parent = find(subsets, subsets[i].parent);
    // 返回根节点
    return subsets[i].parent;
}

// 合并操作（按秩合并）
void Union(Subset subsets[], int x, int y) {
    // 获取x和y的根
    int xroot = find(subsets, x);
    int yroot = find(subsets, y);
    
    // 如果秩不同，将低秩树挂到高秩树下
    if (subsets[xroot].rank < subsets[yroot].rank)
        subsets[xroot].parent = yroot;
    else if (subsets[xroot].rank > subsets[yroot].rank)
        subsets[yroot].parent = xroot;
    else {
        // 秩相同，任选一个作为根，并增加其秩
        subsets[yroot].parent = xroot;
        subsets[xroot].rank++;
    }
}

// 比较函数，用于qsort
int compare(const void* a, const void* b) {
    // 强制转换为Edge指针
    Edge* a1 = (Edge*)a;
    Edge* b1 = (Edge*)b;
    // 按权重升序排列
    return a1->weight > b1->weight;
}

// Kruskal算法主函数
void kruskal_mst(Graph* graph) {
    // 获取顶点和边数量
    int V = graph->V;
    Edge result[V]; // 存储MST的边
    int e = 0; // MST中的边计数器
    int i = 0; // 排序后边的索引
    
    // 对所有边按权重排序
    qsort(graph->edge, graph->E, sizeof(graph->edge[0]), compare);
    
    // 分配并查集内存
    Subset* subsets = (Subset*) malloc(V * sizeof(Subset));
    
    // 初始化并查集
    for (int v = 0; v < V; ++v) {
        subsets[v].parent = v;
        subsets[v].rank = 0;
    }
    
    // 处理所有边
    while (e < V - 1 && i < graph->E) {
        // 获取当前最小权重边
        Edge next_edge = graph->edge[i++];
        
        // 查找源点和目标点的根
        int x = find(subsets, next_edge.src);
        int y = find(subsets, next_edge.dest);
        
        // 如果不会形成环
        if (x != y) {
            // 将边加入MST
            result[e++] = next_edge;
            // 合并两个集合
            Union(subsets, x, y);
        }
    }
    
    // 打印MST
    printf("最小生成树的边:\n");
    for (i = 0; i < e; ++i)
        printf("%d -- %d == %d\n", result[i].src, result[i].dest, result[i].weight);
    
    // 释放内存
    free(subsets);
}

int main() {
    // 创建图（4个顶点，5条边）
    int V = 4;
    int E = 5;
    Graph* graph = create_graph(V, E);
    
    // 添加边
    graph->edge[0].src = 0;
    graph->edge[0].dest = 1;
    graph->edge[0].weight = 10;
    
    graph->edge[1].src = 0;
    graph->edge[1].dest = 2;
    graph->edge[1].weight = 6;
    
    graph->edge[2].src = 0;
    graph->edge[2].dest = 3;
    graph->edge[2].weight = 5;
    
    graph->edge[3].src = 1;
    graph->edge[3].dest = 3;
    graph->edge[3].weight = 15;
    
    graph->edge[4].src = 2;
    graph->edge[4].dest = 3;
    graph->edge[4].weight = 4;
    
    // 执行Kruskal算法
    kruskal_mst(graph);
    
    // 释放图内存
    free(graph->edge);
    free(graph);
    
    return 0;
}
```

Kruskal算法的时间复杂度主要由排序决定，为O(E log E)，其中E是边的数量。由于E最大为V²，所以也可以表示为O(E log V)。该算法特别适合稀疏图，即边数远小于顶点数平方的情况。

### 12.4 最短路径

最短路径问题是图论中的经典问题，旨在找到从源点到其他各顶点的最短路径。这个问题在导航系统、网络路由等实际应用中有着广泛的应用。最短路径可以分为单源最短路径和多源最短路径两种类型。

| 功能名称       | 实例调用方法                   | 具体功能、注意事项、必需参数/可选参数                        |
| -------------- | ------------------------------ | ------------------------------------------------------------ |
| 初始化距离数组 | init_dist(dist, src, vertices) | 将源点距离设为0，其他设为无穷大，必需参数：距离数组、源点、顶点数量 |
| 松弛操作       | relax(u, v, weight, dist)      | 尝试通过u更新v的最短距离，必需参数：起始顶点、目标顶点、边权重、距离数组 |

最短路径问题的求解需要考虑图中是否存在负权边。如果存在负权边，某些算法（如Dijkstra算法）可能无法正确工作，这时需要使用能够处理负权边的算法（如Bellman-Ford算法）。

### 12.5 Dijkstra算法

Dijkstra算法是解决单源最短路径问题的经典算法，适用于不含负权边的有向图或无向图。该算法采用贪心策略，每次从未确定最短路径的顶点中选择距离最小的顶点，并用它来更新其他顶点的距离。

| 功能名称         | 实例调用方法                          | 具体功能、注意事项、必需参数/可选参数                        |
| ---------------- | ------------------------------------- | ------------------------------------------------------------ |
| 查找最小距离顶点 | min_distance(dist, spt_set, vertices) | 在未处理顶点中找到距离最小的顶点，必需参数：距离数组、已处理标记数组、顶点数量 |
| 打印最短路径     | print_solution(dist, vertices)        | 输出从源点到各顶点的最短距离，必需参数：距离数组、顶点数量   |

```c
#include <stdio.h>
#include <limits.h>
#include <stdbool.h>

// 定义顶点数量
#define V 9

// 查找最小距离顶点
int min_distance(int dist[], bool spt_set[]) {
    // 初始化最小距离为无穷大
    int min = INT_MAX, min_index;
    
    // 遍历所有顶点
    for (int v = 0; v < V; v++) {
        // 如果顶点未处理且距离小于当前最小值
        if (spt_set[v] == false && dist[v] <= min) {
            // 更新最小值和索引
            min = dist[v];
            min_index = v;
        }
    }
    // 返回最小距离顶点的索引
    return min_index;
}

// 打印最短路径结果
void print_solution(int dist[], int n) {
    printf("顶点\t距离\n");
    // 遍历所有顶点
    for (int i = 0; i < V; i++) {
        // 打印源点到各顶点的最短距离
        printf("%d \t\t %d\n", i, dist[i]);
    }
}

// Dijkstra算法主函数
void dijkstra(int graph[V][V], int src) {
    // 存储从源点到各顶点的最短距离
    int dist[V];
    // 标记顶点是否已处理
    bool spt_set[V];
    
    // 初始化所有距离为无穷大，处理标记为false
    for (int i = 0; i < V; i++) {
        dist[i] = INT_MAX;
        spt_set[i] = false;
    }
    
    // 源点到自身的距离为0
    dist[src] = 0;
    
    // 处理所有顶点
    for (int count = 0; count < V - 1; count++) {
        // 选择未处理顶点中距离最小的
        int u = min_distance(dist, spt_set);
        // 标记为已处理
        spt_set[u] = true;
        
        // 更新相邻顶点的距离
        for (int v = 0; v < V; v++) {
            // 如果存在边且v未处理且通过u到达v的距离更短
            if (!spt_set[v] && graph[u][v] && dist[u] != INT_MAX && 
                dist[u] + graph[u][v] < dist[v]) {
                // 更新距离
                dist[v] = dist[u] + graph[u][v];
            }
        }
    }
    
    // 打印结果
    print_solution(dist, V);
}

int main() {
    // 定义邻接矩阵表示的图
    int graph[V][V] = {
        {0, 4, 0, 0, 0, 0, 0, 8, 0},
        {4, 0, 8, 0, 0, 0, 0, 11, 0},
        {0, 8, 0, 7, 0, 4, 0, 0, 2},
        {0, 0, 7, 0, 9, 14, 0, 0, 0},
        {0, 0, 0, 9, 0, 10, 0, 0, 0},
        {0, 0, 4, 14, 10, 0, 2, 0, 0},
        {0, 0, 0, 0, 0, 2, 0, 1, 6},
        {8, 11, 0, 0, 0, 0, 1, 0, 7},
        {0, 0, 2, 0, 0, 0, 6, 7, 0}
    };
    
    // 从顶点0开始计算最短路径
    dijkstra(graph, 0);
    
    return 0;
}
```

Dijkstra算法的时间复杂度为O(V²)，其中V是顶点数量。如果使用优先队列（如二叉堆）优化，可以将时间复杂度降低到O((V + E) log V)，其中E是边的数量。需要注意的是，该算法不能处理包含负权边的图，因为负权边可能导致已经确定的最短路径被后续更新。
