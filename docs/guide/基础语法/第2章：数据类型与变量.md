## 2.1 基本数据类型（int、float、double、char）

在C语言的世界里，数据类型就像是我们生活中的各种容器，有的装整数，有的装小数，有的装字符。理解这些基本数据类型是学习C语言的第一步，就像学做菜要先认识锅碗瓢盆一样。

### 基本数据类型详解

#### 整型（int）

整型用于存储整数，就像我们日常生活中的计数一样。C语言提供了多种整型来适应不同的需求。

```c
#include <stdio.h>

int main() {
    // 声明一个int类型的变量，存储整数
    int age = 25;  // int通常占用4个字节，范围：-2147483648 到 2147483647
    
    // 声明一个short类型的变量，存储较小的整数
    short count = 100;  // short通常占用2个字节，范围：-32768 到 32767
    
    // 声明一个long类型的变量，存储较大的整数
    long population = 1400000000L;  // long通常占用4或8个字节
    
    // 声明一个long long类型的变量，存储非常大的整数
    long long big_number = 9223372036854775807LL;  // long long占用8个字节
    
    // 输出各个变量的值
    printf("年龄: %d\n", age);  // %d用于输出int类型
    printf("计数: %hd\n", count);  // %hd用于输出short类型
    printf("人口: %ld\n", population);  // %ld用于输出long类型
    printf("大数字: %lld\n", big_number);  // %lld用于输出long long类型
    
    return 0;
}
```

#### 浮点型（float和double）

浮点型用于存储带小数点的数字，就像我们计算价格、测量距离时需要的精度。

```c
#include <stdio.h>

int main() {
    // 声明一个float类型的变量，单精度浮点数
    float price = 3.14f;  // float占用4个字节，精度约6-7位小数
    
    // 声明一个double类型的变量，双精度浮点数
    double pi = 3.1415926535;  // double占用8个字节，精度约15-16位小数
    
    // 声明一个long double类型的变量，扩展精度浮点数
    long double precise_pi = 3.14159265358979323846L;  // long double占用10或16个字节
    
    // 输出各个变量的值
    printf("价格: %.2f元\n", price);  // %.2f表示保留2位小数
    printf("圆周率: %.10f\n", pi);  // %.10f表示保留10位小数
    printf("精确圆周率: %.20Lf\n", precise_pi);  // %Lf用于输出long double
    
    // 演示浮点数的精度问题
    float a = 0.1f;
    float b = 0.2f;
    float sum = a + b;
    printf("0.1 + 0.2 = %.10f\n", sum);  // 可能不是精确的0.3，这是浮点数的特性
    
    return 0;
}
```

#### 字符型（char）

字符型用于存储单个字符，就像我们写字时的一个个字母或汉字。

```c
#include <stdio.h>

int main() {
    // 声明一个char类型的变量，存储单个字符
    char grade = 'A';  // char占用1个字节，存储ASCII字符
    
    // 声明一个char类型的变量，存储数字字符
    char digit = '5';  // 注意：'5'是字符，不是数字5
    
    // 声明一个char类型的变量，存储转义字符
    char newline = '\n';  // \n表示换行符
    
    // 声明一个char类型的变量，存储ASCII码值
    char ascii_char = 65;  // 65对应的ASCII字符是'A'
    
    // 输出各个变量的值
    printf("成绩等级: %c\n", grade);  // %c用于输出字符
    printf("数字字符: %c\n", digit);
    printf("ASCII码65对应的字符: %c\n", ascii_char);
    
    // 演示字符的算术运算
    char next_char = grade + 1;  // 'A' + 1 = 'B'
    printf("下一个字符: %c\n", next_char);
    
    // 输出字符的ASCII码值
    printf("字符'A'的ASCII码: %d\n", grade);  // 用%d输出字符会显示其ASCII码
    
    return 0;
}
```

### 数据类型大小和范围查询

```c
#include <stdio.h>
#include <limits.h>  // 包含整型范围的宏定义
#include <float.h>   // 包含浮点型范围的宏定义

int main() {
    // 查询各种数据类型的大小（字节数）
    printf("=== 数据类型大小 ===\n");
    printf("char: %zu 字节\n", sizeof(char));
    printf("short: %zu 字节\n", sizeof(short));
    printf("int: %zu 字节\n", sizeof(int));
    printf("long: %zu 字节\n", sizeof(long));
    printf("long long: %zu 字节\n", sizeof(long long));
    printf("float: %zu 字节\n", sizeof(float));
    printf("double: %zu 字节\n", sizeof(double));
    printf("long double: %zu 字节\n", sizeof(long double));
    
    printf("\n=== 整型范围 ===\n");
    printf("char范围: %d 到 %d\n", CHAR_MIN, CHAR_MAX);
    printf("short范围: %d 到 %d\n", SHRT_MIN, SHRT_MAX);
    printf("int范围: %d 到 %d\n", INT_MIN, INT_MAX);
    printf("long范围: %ld 到 %ld\n", LONG_MIN, LONG_MAX);
    
    printf("\n=== 浮点型范围 ===\n");
    printf("float精度位数: %d\n", FLT_DIG);
    printf("float最小值: %e\n", FLT_MIN);
    printf("float最大值: %e\n", FLT_MAX);
    printf("double精度位数: %d\n", DBL_DIG);
    printf("double最小值: %e\n", DBL_MIN);
    printf("double最大值: %e\n", DBL_MAX);
    
    return 0;
}
```

### 无符号整型

除了有符号整型，C语言还提供了无符号整型，用于存储非负整数，范围更大。

```c
#include <stdio.h>
#include <limits.h>

int main() {
    // 声明无符号整型变量
    unsigned char age = 25;  // 范围：0 到 255
    unsigned short count = 65000;  // 范围：0 到 65535
    unsigned int population = 4000000000U;  // 范围：0 到 4294967295
    unsigned long long big_num = 18446744073709551615ULL;  // 最大无符号整数
    
    // 输出无符号整型
    printf("年龄: %u\n", age);  // %u用于输出unsigned int
    printf("计数: %hu\n", count);  // %hu用于输出unsigned short
    printf("人口: %lu\n", population);  // %lu用于输出unsigned long
    printf("大数字: %llu\n", big_num);  // %llu用于输出unsigned long long
    
    // 演示无符号整型的溢出
    unsigned int num = 4294967295;  // 最大值
    printf("最大值: %u\n", num);
    num = num + 1;  // 溢出后变为0
    printf("溢出后: %u\n", num);
    
    return 0;
}
```

### 常用数据类型修饰符

C语言提供了一些修饰符来改变基本数据类型的特性。

```c
#include <stdio.h>

int main() {
    // signed修饰符（默认，通常省略）
    signed int temperature = -10;  // 明确表示有符号整型
    printf("温度: %d°C\n", temperature);
    
    // unsigned修饰符
    unsigned int count = 100;  // 无符号整型
    printf("计数: %u\n", count);
    
    // short和long修饰符
    short int small_num = 100;  // 短整型
    long int big_num = 1000000L;  // 长整型
    printf("小数字: %hd\n", small_num);
    printf("大数字: %ld\n", big_num);
    
    // long long修饰符
    long long very_big_num = 1000000000000LL;  // 很长的整型
    printf("超大数字: %lld\n", very_big_num);
    
    return 0;
}
```

### 数据类型选择建议

| 使用场景           | 推荐数据类型 | 理由               |
| ------------------ | ------------ | ------------------ |
| 年龄、数量等小整数 | int或short   | 节省空间，足够使用 |
| 大数值计算         | long long    | 避免溢出           |
| 价格、测量值       | double       | 精度高             |
| 内存敏感场景       | float        | 占用空间小         |
| 字符处理           | char         | 专门用于字符       |
| 非负数计数         | unsigned     | 范围更大           |

### 注意事项

1. **数据类型的选择**：选择合适的数据类型可以节省内存并提高程序效率。不要一味使用最大的类型。
2. **溢出问题**：整型有固定的范围，超出范围会发生溢出，导致不可预期的结果。
3. **浮点数精度**：浮点数运算存在精度问题，不要用==直接比较两个浮点数是否相等。
4. **字符与字符串**：char存储单个字符，字符串需要用字符数组或指针表示。
5. **平台差异**：不同平台和编译器下，某些数据类型的大小可能不同，使用sizeof查询更可靠。
6. **类型后缀**：使用f、L、LL等后缀可以明确常量的类型，避免隐式转换。
