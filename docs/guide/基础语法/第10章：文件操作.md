## 10.1 文件的打开与关闭（fopen、fclose）

在C语言中，文件操作是程序与外部世界交互的重要方式。无论是读取配置文件、保存用户数据，还是处理日志信息，都离不开文件操作。而文件操作的第一步，就是打开和关闭文件。

### 实例方法表格

| 功能名称 | 实例调用方法                            | 具体功能与注意事项                                           |
| -------- | --------------------------------------- | ------------------------------------------------------------ |
| 打开文件 | `FILE *fp = fopen("filename", "mode");` | mode参数决定文件打开模式（"r"读、"w"写、"a"追加等），返回文件指针，失败返回NULL |
| 关闭文件 | `int result = fclose(fp);`              | 关闭文件指针指向的文件，成功返回0，失败返回EOF               |

### 使用示例

```c
#include <stdio.h>  // 包含标准输入输出头文件，提供fopen和fclose函数声明

int main() {
    FILE *file_ptr;  // 声明文件指针变量，用于指向打开的文件
    
    // 尝试以只读模式打开文件
    file_ptr = fopen("example.txt", "r");  // "r"表示只读模式，如果文件不存在则返回NULL
    
    // 检查文件是否成功打开
    if (file_ptr == NULL) {  // 如果fopen返回NULL，说明文件打开失败
        printf("错误：无法打开文件 example.txt\n");  // 输出错误信息
        return 1;  // 程序异常退出，返回错误码1
    }
    
    printf("文件 example.txt 成功打开！\n");  // 文件打开成功提示
    
    // 关闭文件
    int close_result = fclose(file_ptr);  // 关闭文件，释放相关资源
    
    // 检查文件是否成功关闭
    if (close_result == EOF) {  // 如果fclose返回EOF，说明文件关闭失败
        printf("警告：文件关闭失败\n");  // 输出警告信息
        return 1;  // 程序异常退出
    }
    
    printf("文件已成功关闭。\n");  // 文件关闭成功提示
    return 0;  // 程序正常退出
}
```

### 注意事项

- **必须检查返回值**：`fopen()`可能因为文件不存在、权限不足等原因失败，务必检查返回值是否为NULL
- **及时关闭文件**：打开的文件应该及时关闭，避免资源泄漏
- **文件模式选择**：不同的打开模式有不同的行为，比如"w"模式会清空原有文件内容
- **跨平台兼容性**：在Windows和Linux系统中，文件路径分隔符不同（\ vs /）

小结：文件的打开与关闭是C语言文件操作的基础，通过`fopen()`和`fclose()`函数可以安全地管理文件资源。正确处理这些基本操作是进行后续文件读写的前提。

## 10.2 文件的读写操作（fread、fwrite、fprintf、fscanf）

掌握了文件的打开与关闭后，我们就可以进行实际的文件读写操作了。C语言提供了多种文件读写函数，适用于不同的数据类型和使用场景。

### 实例方法表格

| 功能名称   | 实例调用方法                           | 具体功能与注意事项                                  |
| ---------- | -------------------------------------- | --------------------------------------------------- |
| 格式化写入 | `fprintf(fp, "格式字符串", 参数列表);` | 类似printf，但输出到文件，适用于文本文件写入        |
| 格式化读取 | `fscanf(fp, "格式字符串", &变量列表);` | 类似scanf，但从文件读取，注意要传入变量地址         |
| 二进制写入 | `fwrite(buffer, size, count, fp);`     | 写入二进制数据，size为每个元素大小，count为元素个数 |
| 二进制读取 | `fread(buffer, size, count, fp);`      | 读取二进制数据，返回实际读取的元素个数              |

### 使用示例

```c
#include <stdio.h>  // 包含标准输入输出头文件

int main() {
    FILE *file_ptr;
    char text_data[] = "Hello, File Operation!";  // 要写入的文本数据
    int number = 42;  // 要写入的整数数据
    
    // 以写入模式创建并打开文件
    file_ptr = fopen("data.txt", "w");  // "w"模式会创建新文件或覆盖现有文件
    
    if (file_ptr == NULL) {
        printf("错误：无法创建文件 data.txt\n");
        return 1;
    }
    
    // 使用fprintf写入格式化文本
    fprintf(file_ptr, "文本内容: %s\n", text_data);  // 将格式化字符串写入文件
    fprintf(file_ptr, "数字内容: %d\n", number);     // 将整数写入文件
    
    // 关闭文件
    fclose(file_ptr);
    
    // 重新以读取模式打开文件
    file_ptr = fopen("data.txt", "r");
    if (file_ptr == NULL) {
        printf("错误：无法读取文件 data.txt\n");
        return 1;
    }
    
    char read_text[100];   // 存储读取的文本
    int read_number;       // 存储读取的数字
    
    // 使用fscanf从文件读取格式化数据
    fscanf(file_ptr, "文本内容: %[^\n]", read_text);  // 读取直到换行符的文本
    fscanf(file_ptr, "\n数字内容: %d", &read_number); // 读取整数，注意跳过换行符
    
    printf("从文件读取的内容:\n");
    printf("文本: %s\n", read_text);
    printf("数字: %d\n", read_number);
    
    fclose(file_ptr);
    return 0;
}
```

### 注意事项

- **格式字符串匹配**：`fscanf`的格式字符串必须与文件中的数据格式完全匹配
- **缓冲区溢出风险**：使用`fscanf`读取字符串时要注意指定最大长度，避免缓冲区溢出
- **二进制vs文本模式**：在Windows系统中，二进制模式（"rb"、"wb"）和文本模式的行为不同
- **返回值检查**：`fread`和`fwrite`返回实际处理的元素数量，应检查是否与预期一致

小结：C语言提供了丰富的文件读写函数，`fprintf`和`fscanf`适合处理格式化的文本数据，而`fread`和`fwrite`则适合处理二进制数据。根据具体需求选择合适的函数是高效文件操作的关键。

## 10.3 文件定位与随机访问

在处理大型文件时，我们往往不需要从头到尾顺序读取所有内容，而是需要直接跳转到特定位置进行读写操作。C语言通过文件定位函数支持这种随机访问模式。

### 实例方法表格

| 功能名称     | 实例调用方法                 | 具体功能与注意事项                                           |
| ------------ | ---------------------------- | ------------------------------------------------------------ |
| 获取当前位置 | `long pos = ftell(fp);`      | 返回当前文件指针相对于文件开头的字节偏移量                   |
| 设置文件位置 | `fseek(fp, offset, origin);` | origin可选SEEK_SET(开头)、SEEK_CUR(当前位置)、SEEK_END(结尾) |
| 重置到开头   | `rewind(fp);`                | 等价于fseek(fp, 0L, SEEK_SET)，将文件指针重置到文件开头      |
| 检查文件结束 | `feof(fp);`                  | 检查是否到达文件末尾，返回非0表示已到末尾                    |

### 使用示例

```c
#include <stdio.h>  // 包含标准输入输出头文件

int main() {
    FILE *file_ptr;
    char buffer[100];
    
    // 创建一个包含多行文本的文件
    file_ptr = fopen("random_access.txt", "w");
    if (file_ptr == NULL) {
        printf("错误：无法创建文件\n");
        return 1;
    }
    
    // 写入多行文本
    fprintf(file_ptr, "第一行：这是文件的开始\n");
    fprintf(file_ptr, "第二行：这是中间的内容\n");
    fprintf(file_ptr, "第三行：这是文件的结尾\n");
    fclose(file_ptr);
    
    // 以读取模式重新打开文件
    file_ptr = fopen("random_access.txt", "r");
    if (file_ptr == NULL) {
        printf("错误：无法打开文件\n");
        return 1;
    }
    
    // 获取文件当前位置（应该是0）
    long current_pos = ftell(file_ptr);  // ftell返回当前文件指针位置
    printf("初始位置: %ld 字节\n", current_pos);
    
    // 读取第一行
    fgets(buffer, sizeof(buffer), file_ptr);  // 读取一行文本
    printf("第一行: %s", buffer);
    
    // 记录第二行的起始位置
    long second_line_pos = ftell(file_ptr);  // 记录当前位置，即第二行开始位置
    printf("第二行位置: %ld 字节\n", second_line_pos);
    
    // 读取第二行
    fgets(buffer, sizeof(buffer), file_ptr);
    printf("第二行: %s", buffer);
    
    // 使用fseek跳转到第二行开始位置
    fseek(file_ptr, second_line_pos, SEEK_SET);  // SEEK_SET表示从文件开头计算偏移量
    printf("跳转回第二行位置\n");
    
    // 再次读取第二行
    fgets(buffer, sizeof(buffer), file_ptr);
    printf("重新读取第二行: %s", buffer);
    
    // 跳转到文件末尾
    fseek(file_ptr, 0L, SEEK_END);  // SEEK_END表示从文件末尾计算偏移量
    long file_size = ftell(file_ptr);  // 获取文件总大小
    printf("文件总大小: %ld 字节\n", file_size);
    
    // 重置到文件开头
    rewind(file_ptr);  // 等价于fseek(file_ptr, 0L, SEEK_SET)
    printf("重置到文件开头\n");
    
    fclose(file_ptr);
    return 0;
}
```

### 注意事项

- **偏移量单位**：`fseek`的偏移量是以字节为单位的
- **文本文件限制**：在某些系统上，对文本文件使用`fseek`可能有特殊限制
- **二进制文件推荐**：随机访问通常在二进制文件上效果更好，因为文本文件可能存在换行符转换等问题
- **位置有效性**：确保`fseek`的目标位置在文件有效范围内，避免未定义行为

小结：文件定位函数如`fseek`、`ftell`和`rewind`使得程序能够高效地进行随机文件访问，这对于处理大型数据文件或实现数据库等应用至关重要。掌握这些函数可以显著提升文件操作的灵活性和效率。

## 10.4 常见文件操作错误处理

在实际的文件操作中，各种错误情况都可能发生。良好的错误处理不仅能提高程序的健壮性，还能帮助用户理解问题所在并采取相应措施。

### 实例方法表格

| 功能名称     | 实例调用方法        | 具体功能与注意事项                        |
| ------------ | ------------------- | ----------------------------------------- |
| 检查文件结束 | `feof(fp)`          | 检测是否到达文件末尾，返回非0表示已到末尾 |
| 检查错误标志 | `ferror(fp)`        | 检测文件流是否有错误，返回非0表示有错误   |
| 清除错误标志 | `clearerr(fp)`      | 清除文件流的错误和EOF标志                 |
| 获取错误信息 | `perror("message")` | 打印系统错误信息，基于errno全局变量       |

### 使用示例

```c
#include <stdio.h>  // 标准输入输出头文件
#include <errno.h>  // 错误号头文件，提供errno变量
#include <string.h> // 字符串处理头文件，提供strerror函数

int main() {
    FILE *file_ptr;
    char buffer[100];
    int read_count = 0;
    
    // 尝试打开一个可能不存在的文件
    file_ptr = fopen("nonexistent.txt", "r");
    
    // 检查文件打开是否失败
    if (file_ptr == NULL) {
        // 使用perror打印详细的错误信息
        perror("错误：打开文件失败");  // perror会自动附加系统错误描述
        
        // 或者使用strerror获取错误描述字符串
        printf("详细错误信息: %s\n", strerror(errno));  // errno包含具体的错误代码
        
        // 常见的errno值：
        // ENOENT (2): 文件或目录不存在
        // EACCES (13): 权限不足
        // EISDIR (21): 试图以普通文件方式打开目录
        
        return 1;
    }
    
    // 模拟读取文件内容
    while (fgets(buffer, sizeof(buffer), file_ptr) != NULL) {
        read_count++;
        printf("第%d行: %s", read_count, buffer);
        
        // 检查读取过程中是否发生错误
        if (ferror(file_ptr)) {
            printf("警告：读取过程中发生错误\n");
            clearerr(file_ptr);  // 清除错误标志，尝试继续
        }
    }
    
    // 检查是否是因为到达文件末尾而退出循环
    if (feof(file_ptr)) {
        printf("成功读取到文件末尾，共读取%d行\n", read_count);
    } else {
        // 如果不是因为文件末尾，可能是读取错误
        printf("读取异常终止，可能发生了I/O错误\n");
    }
    
    // 尝试关闭文件并检查关闭结果
    if (fclose(file_ptr) != 0) {
        perror("警告：关闭文件时发生错误");
    }
    
    return 0;
}
```

### 注意事项

- **及时检查错误**：每次文件操作后都应该检查是否成功
- **errno的时效性**：`errno`只在函数调用失败时才有意义，成功调用可能会重置它
- **区分EOF和错误**：读取函数返回NULL可能是因为到达文件末尾，也可能是因为发生错误，需要用`feof()`和`ferror()`区分
- **错误恢复策略**：根据错误类型决定是重试、跳过还是终止程序
- **用户友好性**：向用户提供清晰、有用的错误信息，而不是技术性的错误代码
