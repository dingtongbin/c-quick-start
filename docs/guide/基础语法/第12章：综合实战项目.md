```c
// 添加学生信息
int addStudent(Student *stu, int count) {
    // 临时变量存储输入
    char tempId[20];
    char tempName[50];
    int tempAge;
    float tempMath, tempEnglish, tempComputer;
    
    // 输入学号
    printf("请输入学号: ");
    scanf("%s", tempId);
    
    // 检查学号是否已存在
    for(int i = 0; i < count; i++) {
        if(strcmp(stu[i].id, tempId) == 0) {
            printf("该学号已存在！\n");
            return 0;
        }
    }
    
    // 输入姓名
    printf("请输入姓名: ");
    scanf("%s", tempName);
    
    // 输入年龄
    printf("请输入年龄: ");
    if(scanf("%d", &tempAge) != 1 || tempAge <= 0 || tempAge > 150) {
        printf("年龄输入无效！\n");
        while(getchar() != '\n'); // 清空缓冲区
        return 0;
    }
    
    // 输入数学成绩
    printf("请输入数学成绩 (0-100): ");
    if(scanf("%f", &tempMath) != 1 || tempMath < 0 || tempMath > 100) {
        printf("成绩输入无效！\n");
        while(getchar() != '\n');
        return 0;
    }
    
    // 输入英语成绩
    printf("请输入英语成绩 (0-100): ");
    if(scanf("%f", &tempEnglish) != 1 || tempEnglish < 0 || tempEnglish > 100) {
        printf("成绩输入无效！\n");
        while(getchar() != '\n');
        return 0;
    }
    
    // 输入计算机成绩
    printf("请输入计算机成绩 (0-100): ");
    if(scanf("%f", &tempComputer) != 1 || tempComputer < 0 || tempComputer > 100) {
        printf("成绩输入无效！\n");
        while(getchar() != '\n');
        return 0;
    }
    
    // 计算平均分
    float avg = (tempMath + tempEnglish + tempComputer) / 3.0;
    
    // 保存到结构体
    strcpy(stu[count].id, tempId);
    strcpy(stu[count].name, tempName);
    stu[count].age = tempAge;
    stu[count].math = tempMath;
    stu[count].english = tempEnglish;
    stu[count].computer = tempComputer;
    stu[count].average = avg;
    
    return 1; // 成功
}
```

```c
// 保存到文件
void saveToFile(Student *stu, int count) {
    FILE *fp = fopen("students.dat", "wb");
    
    if(fp == NULL) {
        printf("打开文件失败！\n");
        return;
    }
    
    // 写入学生数量
    fwrite(&count, sizeof(int), 1, fp);
    
    // 写入学生数据
    for(int i = 0; i < count; i++) {
        fwrite(&stu[i], sizeof(Student), 1, fp);
    }
    
    fclose(fp);
    printf("数据已保存到 students.dat\n");
}

// 从文件加载
int loadFromFile(Student *stu) {
    FILE *fp = fopen("students.dat", "rb");
    
    if(fp == NULL) {
        printf("文件不存在，将创建新文件。\n");
        return 0;
    }
    
    int count = 0;
    
    // 读取学生数量
    if(fread(&count, sizeof(int), 1, fp) != 1) {
        fclose(fp);
        printf("读取文件失败！\n");
        return 0;
    }
    
    // 读取学生数据
    for(int i = 0; i < count; i++) {
        if(fread(&stu[i], sizeof(Student), 1, fp) != 1) {
            fclose(fp);
            printf("读取数据失败！\n");
            return i; // 返回已成功读取的数量
        }
    }
    
    fclose(fp);
    printf("成功加载 %d 条记录。\n", count);
    return count;
}
```

```c
int addContact(Contact **contacts, int *count) {

    // 输入备注
```

```c
// 删除联系人
int deleteContact(Contact **contacts, int *count, char *name) {
    int index = -1;
    
    // 查找要删除的联系人
    for(int i = 0; i < *count; i++) {
        if(strcmp((*contacts)[i].name, name) == 0) {
            index = i;
            break;
        }
    }
    
    if(index == -1) {
        printf("未找到该联系人！\n");
        return 0;
    }
    
    // 确认删除
    char confirm;
    printf("确定要删除 %s 吗? (y/n): ", name);
    scanf(" %c", &confirm);
    
    if(confirm != 'y' && confirm != 'Y') {
        printf("取消删除。\n");
        return 0;
    }
    
    // 移动后续元素覆盖要删除的元素
    for(int i = index; i < *count - 1; i++) {
        (*contacts)[i] = (*contacts)[i + 1];
    }
    
    (*count)--;
    
    // 如果容量使用率过低，可以考虑缩容
    if(*count > 0 && *count < (*count + INITIAL_CAPACITY) / 2) {
        Contact *temp = realloc(*contacts, *count * sizeof(Contact));
        if(temp != NULL) {
            *contacts = temp;
            printf("通讯录容量已优化。\n");
        }
    }
    
    printf("删除成功！\n");
    return 1;
}
```

```c
// 保存通讯录到文件
void saveContacts(Contact *contacts, int count) {

// 从文件加载通讯录
int loadContacts(Contact **contacts, int *count) {
    FILE *fp = fopen("contacts.txt", "r");
    
    if(fp == NULL) {
        printf("通讯录文件不存在，将创建新的通讯录。\n");
        *contacts = malloc(INITIAL_CAPACITY * sizeof(Contact));
        if(*contacts == NULL) {
            printf("内存分配失败！\n");
            return 0;
        }
        *count = 0;
        return 0;
    }
    
    // 读取联系人数量
    fscanf(fp, "%d\n", count);
    
    if(*count <= 0) {
        fclose(fp);
        *contacts = malloc(INITIAL_CAPACITY * sizeof(Contact));
        *count = 0;
        return 0;
    }
    
    // 分配内存
    *contacts = malloc((*count + INITIAL_CAPACITY) * sizeof(Contact));
    if(*contacts == NULL) {
        printf("内存分配失败！\n");
        fclose(fp);
        return 0;
    }
    
    // 读取每个联系人的信息
    for(int i = 0; i < *count; i++) {
        char line[300];
        if(fgets(line, sizeof(line), fp) == NULL) {
            printf("读取文件出错！\n");
            fclose(fp);
            free(*contacts);
            *count = 0;
            return 0;
        }
        
        // 去掉换行符
        line[strcspn(line, "\n")] = 0;
        
        // 解析字段
        char *token = strtok(line, "|");
        if(token != NULL) strcpy((*contacts)[i].name, token);
        
        token = strtok(NULL, "|");
        if(token != NULL) strcpy((*contacts)[i].phone, token);
        
        token = strtok(NULL, "|");
        if(token != NULL) strcpy((*contacts)[i].email, token);
        
        token = strtok(NULL, "|");
        if(token != NULL) strcpy((*contacts)[i].address, token);
        
        token = strtok(NULL, "|");
        if(token != NULL) strcpy((*contacts)[i].remark, token);
    }
    
    fclose(fp);
    printf("成功加载 %d 个联系人。\n", *count);
    return *count;
}
```

```c
// 计算两个数的运算结果
double calculate(double a, char op, double b) {
    switch(op) {
        case '+':
            return a + b;
        case '-':
            return a - b;
        case '*':
            return a * b;
        case '/':
            if(b == 0) {
                printf("错误：除数不能为零！\n");
                return 0;
            }
            return a / b;
        default:
            printf("错误：未知运算符 %c\n", op);
            return 0;
    }
}

// 跳过空白字符
void skipWhitespace(char **expr) {
    while(**expr && isspace(**expr)) {
        (*expr)++;
    }
}

// 获取数字（支持浮点数）
double getNumber(char **expr) {
    skipWhitespace(expr);
    
    char *endptr;
    double num = strtod(*expr, &endptr);
    
    if(endptr == *expr) {
        printf("错误：期望数字，但找到了 '%c'\n", **expr);
        return 0;
    }
    
    *expr = endptr;
    return num;
}

// 获取运算符
char getOperator(char **expr) {
    skipWhitespace(expr);
    
    if(**expr == '\0') {
        return '\0';
    }
    
    char op = **expr;
    (*expr)++;
    return op;
}
```

```c
// 计算表达式（支持运算符优先级）
double evaluateExpression(char *expr) {
    if(expr == NULL || *expr == '\0') {
        printf("错误：空表达式！\n");
        return 0;
    }
    
    char *pos = expr;
    double result = getNumber(&pos);
    
    while(*pos != '\0') {
        char op = getOperator(&pos);
        
        if(op == '\0') {
            break;
        }
        
        double nextNum = getNumber(&pos);
        
        // 处理乘除优先级
        if(op == '*' || op == '/') {
            result = calculate(result, op, nextNum);
        } else if(op == '+' || op == '-') {
            // 需要查看下一个运算符
            char *lookahead = pos;
            skipWhitespace(&lookahead);
            
            if(*lookahead == '*' || *lookahead == '/') {
                // 下一个运算是乘除，先计算后面的
                char nextOp = getOperator(&pos);
                double nextNextNum = getNumber(&pos);
                nextNum = calculate(nextNum, nextOp, nextNextNum);
            }
            
            result = calculate(result, op, nextNum);
        } else {
            printf("错误：无效的运算符 '%c'\n", op);
            return 0;
        }
    }
    
    return result;
}
```

```c
// 验证表达式基本语法
int validateExpression(char *expr) {
    int parenCount = 0;
    int hasDigit = 0;
    
    for(int i = 0; expr[i] != '\0'; i++) {
        char c = expr[i];
        
        if(isdigit(c) || c == '.') {
            hasDigit = 1;
        } else if(c == '(') {
            parenCount++;
        } else if(c == ')') {
            parenCount--;
            if(parenCount < 0) {
                printf("错误：括号不匹配！\n");
                return 0;
            }
        } else if(!isspace(c) && c != '+' && c != '-' && c != '*' && c != '/') {
            printf("错误：无效字符 '%c'\n", c);
            return 0;
        }
    }
    
    if(parenCount != 0) {
        printf("错误：括号不匹配！\n");
        return 0;
    }
    
    if(!hasDigit) {
        printf("错误：表达式中没有数字！\n");
        return 0;
    }
    
    return 1;
}
```

```c
double parseExpression(char **expr) {
    double result = parseTerm(expr);
    
    while(**expr) {
        skipWhitespace(expr);
        
        if(**expr == '+') {
            (*expr)++;
            result += parseTerm(expr);
        } else if(**expr == '-') {
            (*expr)++;
            result -= parseTerm(expr);
        } else {
            break;
        }
    }
    
    return result;
}

// 解析项（处理 * 和 /）
double parseTerm(char **expr) {
    double result = parseFactor(expr);
    
    while(**expr) {
        skipWhitespace(expr);
        
        if(**expr == '*') {
            (*expr)++;
            result *= parseFactor(expr);
        } else if(**expr == '/') {
            (*expr)++;
            double divisor = parseFactor(expr);
            if(divisor == 0) {
                printf("错误：除数不能为零！\n");
                return result;
            }
            result /= divisor;
        } else {
            break;
        }
    }
    
    return result;
}

// 解析因子（数字或括号表达式）
double parseFactor(char **expr) {
    skipWhitespace(expr);

    if(**expr == '(') {
        (*expr)++;
        double result = parseExpression(expr);
        
        skipWhitespace(expr);
        if(**expr == ')') {
            (*expr)++;
            return result;
        } else {
            printf("错误：缺少右括号！\n");

        }
    }
    
    // 处理数字

}
```

```c
                    return pow(base, exp);

    return parseFactor(expr);
}
```
